def linear_search(arr, target):
    for i, v in enumerate(arr):
        if v == target:
            return i
    return -1
⚡ 이진 탐색 O(log n)
def binary_search(arr, target):
    lo, hi = 0, len(arr) - 1
    while lo <= hi:
        mid = (lo + hi) // 2
        if arr[mid] == target: return mid
        elif arr[mid] < target: lo = mid + 1
        else: hi = mid - 1
    return -1
🌊 BFS — 너비 우선
from collections import deque
def bfs(graph, start):
    visited = set([start])
    q = deque([start])
    while q:
        node = q.popleft()
        for nei in graph[node]:
            if nei not in visited:
                visited.add(nei)
                q.append(nei)
🌿 DFS — 깊이 우선
def dfs(graph, node, visited=None):
    if visited is None:
        visited = set()
    visited.add(node)
    for nei in graph[node]:
        if nei not in visited:
            dfs(graph, nei, visited)
    return visited
🗺 다익스트라 (최단경로)
가중치 있는 그래프, O((V+E) log V)
import heapq
def dijkstra(graph, src):
    dist = {n: float('inf') for n in graph}
    dist[src] = 0
    pq = [(0, src)]
    while pq:
        d, u = heapq.heappop(pq)
        if d > dist[u]: continue
        for v, w in graph[u]:
            if dist[u]+w < dist[v]:
                dist[v] = dist[u]+w
                heapq.heappush(pq,(dist[v],v))
    return dist
🔗 Union-Find (서로소 집합)
parent = list(range(n))
def find(x):
    if parent[x] != x:
        parent[x] = find(parent[x])
    return parent[x]
def union(a, b):
    a, b = find(a), find(b)
    if a != b: parent[b] = a
🌲 위상 정렬 (Topological Sort)
DAG에서 의존성 순서 결정 (과목 선수체계 등)
from collections import deque
def topo_sort(graph, in_deg, n):
    q = deque([i for i in range(n) if in_deg[i]==0])
    order = []
    while q:
        u = q.popleft(); order.append(u)
        for v in graph[u]:
            in_deg[v] -= 1
            if in_deg[v] == 0: q.append(v)
    return order

🔄 프로세스 vs 스레드
프로세스
독립된 메모리 공간 (코드/데이터/힙/스택)
Context Switch 비용 큼
IPC 필요 (파이프, 소켓 등)

스레드
프로세스 내 실행 단위
코드/데이터/힙 공유, 스택은 독립
Context Switch 비용 적음
동기화 필요 (Mutex, Semaphore)
